import time import random class BipolarGame: def __init__(self, name): self.name = name self.energy = 50 self.mood = 50 # 0: Depresyon, 100: Mani self.days = 1 self.game_over = False def check_status(self): print(f"\n--- Gün {self.days} | {self.name}'in Durumu ---") print(f"Enerji: {self.energy} | Ruh Hali: {self.mood}") if self.mood > 90: print("Durum: Manik bir atak geçiriyorsun! Çok hızlı hareket ediyorsun.") elif self.mood < 10: print("Durum: Derin bir depresyon evresindesin. Yataktan kalkmak zor.") else: print("Durum: Dengeli seyrediyor.") def play_turn(self): print("\nNe yapmak istersin?") print("1. Yoğun bir projeye başla (Enerji harca, Manik tetikleme)") print("2. Dinlen ve kitap oku (Dengeyi koru)") print("3. İnsanlarla sosyalleş (Ruh halini yükselt)") choice = input("> ") if choice == "1": self.energy -= 20 self.mood += 15 print("West: 'Dünyayı değiştireceğim!'") elif choice == "2": self.energy += 10 self.mood -= 5 print("West: 'Sessizlik iyi geliyor.'") elif choice == "3": self.energy -= 5 self.mood += 10 print("West: 'Biraz arkadaşlık iyi geldi.'") self.days += 1 self.validate_state() def validate_state(self): if self.energy <= 0: print("\nEnerjin tükendi! West bir süreliğine çöktü.") self.energy = 20 if self.mood > 120: print("\nÇok yüksek bir manik atak! Kontrolü kaybettin.") self.game_over = True if self.mood < -20: print("\nDepresyon çok ağırlaştı. Hastaneye kaldırıldın.") self.game_over = True # Oyun Başlatıcı west = BipolarGame("West") while not west.game_over: west.check_status() west.play_turn() time.sleep(1)
